  <script>
    function redirectToForm() {
      // Function to handle redirection
      let cookieValue = document.cookie.replace(/(?:(?:^|;\s*)formVersion=([^;]*))/.exec("") || ["", ""])[1];
      if (!cookieValue) {
        // No cookie found, randomly assign version (A or B) and set cookie
        let version = Math.random() < 0.5 ? "A" : "B";
        document.cookie = "formVersion=" + version + "; expires=" + new Date(Date.now() + 86400000).toUTCString() + "; path=/";
        cookieValue = version;
      }
      // Redirect based on cookie value
      if (cookieValue === "A") {
        window.location.href = "https://anationary.github.io/pl/slownik-anatomiczny/"; // Replace "url1" with your actual Version A form URL
      } else if (cookieValue === "B") {
        window.location.href = "https://markdownguide.offshoot.io/hacks/"; // Replace "url2" with your actual Version B form URL
      } else {
        // Handle unexpected cookie value (optional)
        console.error("Unexpected cookie value:", cookieValue);
      }
    }
  </script>
</head>
